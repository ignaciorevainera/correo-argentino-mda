### Task 2: Create category reorder API endpoint for aplicativos

**Files:**
- Create: src/pages/api/admin/aplicativos/reorder-categorias.ts

- [ ] **Step 1: Write the endpoint**

Create src/pages/api/admin/aplicativos/reorder-categorias.ts:

`	ypescript
import type { APIRoute } from "astro";
import { db } from "@db/index";
import { applicationCategories } from "@db/schema";
import { eq } from "drizzle-orm";
import { logAdminAction } from "@lib/auditLogger";

export const POST: APIRoute = async ({ request, locals }) => {
  if (!locals.user) {
    return new Response(JSON.stringify({ error: "No autorizado" }), {
      status: 401,
      headers: { "Content-Type": "application/json" },
    });
  }

  try {
    const { items } = await request.json();

    if (!Array.isArray(items)) {
      return new Response(JSON.stringify({ error: "Datos inválidos" }), {
        status: 400,
        headers: { "Content-Type": "application/json" },
      });
    }

    db.transaction((tx) => {
      for (const item of items) {
        tx.update(applicationCategories)
          .set({ sortOrder: item.sortOrder })
          .where(eq(applicationCategories.id, item.id))
          .run();
      }
    });

    await logAdminAction(
      locals.user.username || "Sistema",
      Reordenó  categorías de aplicativos
    );

    return new Response(JSON.stringify({ success: true }), {
      status: 200,
      headers: { "Content-Type": "application/json" },
    });
  } catch (err: any) {
    console.error("[Reorder Categorias Aplicativos API] Error:", err);
    return new Response(
      JSON.stringify({ error: "Error interno al reordenar categorías" }),
      {
        status: 500,
        headers: { "Content-Type": "application/json" },
      }
    );
  }
};
`

// ----------------------------------------------------
// Drag and Drop: Reorder categories in sidebar
// ----------------------------------------------------
let draggedCategoryItem: HTMLElement | null = null;

const categoryDragItems = Array.from(
  document.querySelectorAll<HTMLElement>(".category-item[data-category-dropzone]")
);

categoryDragItems.forEach(item => {
  item.addEventListener("dragstart", (e) => {
    const catId = item.dataset.categoryId;
    if (catId === "all" || catId === "uncategorized") {
      e.preventDefault();
      return;
    }
    draggedCategoryItem = item;
    item.classList.add("opacity-40");
    if (e.dataTransfer) {
      e.dataTransfer.effectAllowed = "move";
      e.dataTransfer.setData("text/category-id", catId || "");
    }
  });

  item.addEventListener("dragend", () => {
    item.classList.remove("opacity-40");
    draggedCategoryItem = null;
  });

  item.addEventListener("dragover", (e) => {
    e.preventDefault();
    if (!draggedCategoryItem || draggedCategoryItem === item) return;
    const targetCatId = item.dataset.categoryId;
    const draggedCatId = draggedCategoryItem.dataset.categoryId;
    if (targetCatId === "all" || targetCatId === "uncategorized" || draggedCatId === "all" || draggedCatId === "uncategorized") return;

    const bounding = item.getBoundingClientRect();
    const offset = e.clientY - bounding.top - bounding.height / 2;

    const categoriesList = document.getElementById("categories-list");
    if (categoriesList) {
      if (offset < 0) {
        categoriesList.insertBefore(draggedCategoryItem, item);
      } else {
        categoriesList.insertBefore(draggedCategoryItem, item.nextSibling);
      }
    }
  });

  item.addEventListener("drop", (e) => {
    e.preventDefault();
    saveReorderedCategories();
  });
});
`

- [ ] **Step 5: Add saveReorderedCategories function**

After saveReorderedApps, add:

`	ypescript
const saveReorderedCategories = async () => {
  try {
    const categoriesList = document.getElementById("categories-list");
    if (!categoriesList) return;

    const categoryEls = Array.from(
      categoriesList.querySelectorAll<HTMLElement>(".category-item[data-category-dropzone]")
    );

    const items: { id: number; sortOrder: number }[] = [];
    let index = 0;

    categoryEls.forEach(el => {
      const catId = el.dataset.categoryId;
      if (catId === "all" || catId === "uncategorized") return;
      const id = parseInt(catId || "0");
      if (!id) return;
      items.push({ id, sortOrder: index * 10 });
      index++;
    });

    if (items.length === 0) return;

    const base = import.meta.env.BASE_URL || "/";
    const cleanBase = base.endsWith("/") ? base : base + "/";

    const response = await fetch(${cleanBase}api/admin/aplicativos/reorder-categorias, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({ items }),
    });

    if (!response.ok) throw new Error("Error de red");
    (window as any).showToast?.("Categorías reordenadas con éxito", "alert-success");
  } catch (err) {
    console.error(err);
    (window as any).showToast?.("Error al guardar el nuevo orden de categorías", "alert-error");
  }
};
`
